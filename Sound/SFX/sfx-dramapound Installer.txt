
SongTable(SFX_DramaPound,sfx-dramapound,SFXGroup)

    #include "MPlayDef.event"

    #define    loop_sfx_grp_sfx-dramapound  sfx-dramapound_map
    #define    loop_sfx_pri_sfx-dramapound  0
    #define    loop_sfx_rev_sfx-dramapound  0
    #define    loop_sfx_mvl_sfx-dramapound  127
    #define    loop_sfx_key_sfx-dramapound  0
    #define    loop_sfx_tbs_sfx-dramapound  1
    #define    loop_sfx_exg_sfx-dramapound  0
    #define    loop_sfx_cmp_sfx-dramapound  1

    ALIGN 4

loop_sfx_1:
    BYTE        VOL     127*loop_sfx_mvl/mxv
    BYTE    KEYSH   loop_sfx_key_sfx-dramapound+0
    BYTE        VOICE   0
    BYTE        TIE     Cn3   v064
    BYTE    W96
loop_sfx_1_B1:
    BYTE    W96
    BYTE    W96
    BYTE    GoTo
     POIN2    loop_sfx_1_B1
loop_sfx_1_B2:
    BYTE    W06
    BYTE        EOT     Cn3 
    BYTE    FINE

    ALIGN 4

sfx-dramapound:
    BYTE    1    // NumTrks
    BYTE    0    // NumBlks
    BYTE    loop_sfx_pri_sfx-dramapound    // Priority
    BYTE    loop_sfx_rev_sfx-dramapound    // Reverb.

    POIN2    loop_sfx_grp_sfx-dramapound

    POIN2    loop_sfx_1

    
sfx-dramapound_map:
WORD $3c00
POIN sfx-dramapound_wav
WORD $FF00FF

sfx-dramapound_wav:
#incbin "sfx-dramapound.bin"
